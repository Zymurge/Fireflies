  @yMin = yMin; @yMax = yMax
  @rand = Random.new
  @dots = genDots( numDots )

def genColor
  r = @rand.rand( 190..250 ) 
  g = @rand.rand( 160..220 )
  b = @rand.rand( 105..160 )
  color = Gosu::Color.new( 255, r, g, b )
  # puts "genColor: #{color.to_s}"

def genDots( num )
  dots = Array.new
  num.times do
    x = @rand.rand( @xMin...@xMax )
    y = @rand.rand( @yMin...@yMax )
    color = genColor
    pulse = @rand.rand( 40..76 )
    pulseStep = @rand.rand( 0...pulse*2 )
    dots << Dot.new( x, y, color, pulse, pulseStep, 15, 80 )    
  end
  dots
end
