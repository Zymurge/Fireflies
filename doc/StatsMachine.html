def update
  lit = 0
  dark = 0
  @dots.each do |dot|
    if dot.isLit? then
      lit += 1
    else
      dark += 1
    end
  end
  
  percent = lit < dark ? 100 * dark / @dots.count : 100 * lit / @dots.count
  @ma.push_period( percent )
end
