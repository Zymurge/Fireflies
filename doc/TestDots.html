    Expect that with a pulse of 3 and 50% fade pattern:

    • Step 1; initial color

    • Step 2: 25% reduction in rgb attributes

    • Step 3: 50% reduction in rgb attributes

    • Step 4-6: as step 3

    • Step 7: return to initial

def testColorChange
  @aColor = Gosu::Color.new( 255, 200, 200, 200 )
  @bColor = Gosu::Color.new( 255, 150, 150, 150 )
  @cColor = Gosu::Color.new( 255, 100, 100, 100 )
  d = Dot.new( 10, 20, @aColor, 3, 0, 2, 50 )
  assert_equal( d.color, @aColor )
  d.move
  assert_equal( d.color, @bColor )
  d.move
  assert_equal( d.color, @cColor )   
  d.move
  assert_equal( d.color, @cColor )
  d.move; d.move; d.move
  assert_equal( d.pulseStep, 0, "Pulse should have wrapped around after 3x2 moves")
  assert_equal( d.color, @aColor )       
end

def testInitializeAttributes
  @aColor = Gosu::Color.new( 255, 200, 200, 200 )
  d = Dot.new( 10, 20, @aColor, 25 )
  assert_equal( d.x, 10 )
  assert_equal( d.y, 20 )
  assert_equal( d.color, @aColor )
end
